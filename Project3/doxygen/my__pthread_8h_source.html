230 /*
231  We define my_sigset() and use that instead of the system sigset() so that
232  we can favor an implementation based on sigaction(). On some systems, such
233  as Mac OS X, sigset() results in flags such as SA_RESTART being set, and
234  we want to make sure that no such flags are set.
235 */
236 #if defined(HAVE_SIGACTION) && !defined(my_sigset)
237 #define my_sigset(A,B) do { struct sigaction l_s; sigset_t l_set; int l_rc; \
238  DBUG_ASSERT((A) != 0); \
239  sigemptyset(&l_set); \
240  l_s.sa_handler = (B); \
241  l_s.sa_mask = l_set; \
242  l_s.sa_flags = 0; \
243  l_rc= sigaction((A), &l_s, (struct sigaction *) NULL);\
244  DBUG_ASSERT(l_rc == 0); \
245  } while (0)
246 #elif defined(HAVE_SIGSET) && !defined(my_sigset)
247 #define my_sigset(A,B) sigset((A),(B))
248 #elif !defined(my_sigset)
249 #define my_sigset(A,B) signal((A),(B))
