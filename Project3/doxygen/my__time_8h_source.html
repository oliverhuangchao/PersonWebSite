101 /*
102  Function to check sanity of a TIMESTAMP value
103 
104  DESCRIPTION
105  Check if a given MYSQL_TIME value fits in TIMESTAMP range.
106  This function doesn't make precise check, but rather a rough
107  estimate.
108 
109  RETURN VALUES
110  FALSE The value seems sane
111  TRUE The MYSQL_TIME value is definitely out of range
112 */
113 
114 static inline my_bool validate_timestamp_range(const MYSQL_TIME *t)
115 {
116  if ((t->year > TIMESTAMP_MAX_YEAR || t->year < TIMESTAMP_MIN_YEAR) ||
117  (t->year == TIMESTAMP_MAX_YEAR && (t->month > 1 || t->day > 19)) ||
118  (t->year == TIMESTAMP_MIN_YEAR && (t->month < 12 || t->day < 31)))
119  return FALSE;
120 
121  return TRUE;
122 }
