225 /*
226  We want to use the 'e' format in some cases even if we have enough space
227  for the 'f' one just to mimic sprintf("%.15g") behavior for large integers,
228  and to improve it for numbers < 10^(-4).
229  That is, for |x| < 1 we require |x| >= 10^(-15), and for |x| > 1 we require
230  it to be integer and be <= 10^DBL_DIG for the 'f' format to be used.
231  We don't lose precision, but make cases like "1e200" or "0.00001" look nicer.
232 */
233 #define MAX_DECPT_FOR_F_FORMAT DBL_DIG

361 static inline const uchar *skip_trailing_space(const uchar *ptr,size_t len)
362 {
363  const uchar *end= ptr + len;
364 
365  if (len > 20)
366  {
367  const uchar *end_words= (const uchar *)(intptr)
368  (((ulonglong)(intptr)end) / SIZEOF_INT * SIZEOF_INT);
369  const uchar *start_words= (const uchar *)(intptr)
370  ((((ulonglong)(intptr)ptr) + SIZEOF_INT - 1) / SIZEOF_INT * SIZEOF_INT);
371 
372  DBUG_ASSERT(((ulonglong)(intptr)ptr) >= SIZEOF_INT);
373  if (end_words > ptr)
374  {
375  while (end > end_words && end[-1] == 0x20)
376  end--;
377  if (end[-1] == 0x20 && start_words < end_words)
378  while (end > start_words && ((unsigned *)end)[-1] == SPACE_INT)
379  end -= SIZEOF_INT;
380  }
381  }
382  while (end > ptr && end[-1] == 0x20)
383  end--;
384  return (end);
385 }
